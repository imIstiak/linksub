# Complete Heroku Deployment Commands
# Run these commands one by one in PowerShell

# 1. Login to Heroku (this will open a browser)
heroku login

# 2. Create a new Heroku app (replace 'your-app-name' with your desired name)
# App names must be unique across all Heroku apps
heroku create rmb-bdt-calculator-2025

# Alternative: Let Heroku generate a random name
# heroku create

# 3. Add Heroku remote to your git repository
# This is usually done automatically when you create the app

# 4. Check your remotes
git remote -v

# 5. Deploy to Heroku
git push heroku master

# 6. Open your app in browser
heroku open

# 7. View logs if there are any issues
heroku logs --tail

# 8. Check app status
heroku ps

# Additional useful commands:
# - Restart the app: heroku restart
# - View config vars: heroku config
# - Scale dynos: heroku ps:scale web=1
